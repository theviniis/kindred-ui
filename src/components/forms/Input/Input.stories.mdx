## <a id="usage"></a>Usage


```jsx
export default function App() {
  const [username, setUsername] = React.useState('');

  return (
    <form>
      <Input
        type="text"
        label="Username"
        value={username}
        onChange={e => setUsername(e.target.value)}
      />
    </form>
  );
}
```
